#include "../include/Stack/Stack.hpp"
#include "../include/VM/VirtualMachine.hpp"

int main()
{
    VM vm;

    //allocate memory for int.
    std::pair<int, char> x = vm.allocateMemory("125", 'i');

    //allocate memory for string.
    std::pair<int, char> a = vm.allocateMemory("Matei", 's');

    //allocate memory for float.
    std::pair<int, char> p = vm.allocateMemory("30.21", 'f');

    //print the the address of the variables or the first if it's a string.
    std::cout << x.first << std::endl;
    std::cout << a.first << std::endl;
    std::cout << p.first << std::endl;
    
    //we retrieve the value at block 'x'.
    RetrievedData b = vm.retrieveData(x.first);

    //print the integer.
    std::cout << b << std::endl;

    //print the string.
    for (int i = a.first; i < a.first + 5; i++)
    {
        //we retrieve the blocks of which the string is composed. Starting from 'a'.
        b = vm.retrieveData(i);
        std::cout << b;
    }
    std::cout << std::endl;
    
    //we retrieve the value at block 'p'
    b = vm.retrieveData(p.first);

    //print the float
    std::cout << b << std::endl;

}
